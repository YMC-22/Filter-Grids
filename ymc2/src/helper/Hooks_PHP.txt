Названия фильтов в плагине:
1. Следуй паттерну: {namespace}_{context}_{specific}
Example:
- ymc — префикс плагина (оставляем)
- pagination — контекст (оставляем)
- prev_text — то, что фильтруется (оставляем, но можно улучшить)





### Filters and Action Hooks in PHP ###

1. Pagination (Filters)

Button text Prev
apply_filters('ymc/pagination/prev_text',  $prev_button_text)
apply_filters('ymc/pagination/prev_text_'. $filter_id, $prev_button_text)
apply_filters('ymc/pagination/prev_text_'. $filter_id .'_'. $counter, $prev_button_text)

Button text Next
apply_filters('ymc/pagination/next_text',  $next_button_text)
apply_filters('ymc/pagination/next_text_'. $filter_id, $next_button_text)
apply_filters('ymc/pagination/next_text_'. $filter_id .'_'. $counter, $next_button_text)

6. Filter Custom Layout (Filters)

$placement = top | left | right
$container_class is string

/**
 * Фильтр HTML блока кастомного фильтра.
 *
 * @param string $guide_html HTML контент блока `.filter-custom-guide`.
 * @param int $filter_id ID текущего фильтра.
 * @param array $tax_name Список таксономий (например, ['category']).
 * @param array $term_settings Настройки терминов по таксономиям.
 * @param string $container_class CSS классы контейнера фильтра.
 */




$term_settings = [
    'taxonomy' => 'category',
    'terms' => [
        12 => [
            'term_id'           =>  12, (int)
            'term_name'         =>  'News', (string)
            'term_slug'         =>  'term-news', (string)
            'term_background'   =>  '#095c81', (string)
            'term_color'        =>  '#ffffff', (string)
            'term_class'        =>  'my-class', (string)
            'term_default'      =>  true, (bool)
            'term_visible'      =>  true, (bool)
            'term_checked'      =>  true, (bool)
            'term_icon_class'   =>  'far fa-paper-plane', (string)
            'term_icon_color'   =>  '#ffffff', (string)
            'term_icon_alignment' => 'right-icon' (string)
        ]
    ]
];

apply_filters('ymc/filter/layout/top/custom', $output, $filter_id, $taxonomies, $term_settings, $container_class);
apply_filters('ymc/filter/layout/{$placement}/custom_{$filter_id}', $output, $filter_id, $taxonomies, $term_settings, $container_class);
apply_filters('ymc/filter/layout/{$placement}/custom_{$filter_id}_" . $filter_counter', $output, $filter_id, $taxonomies, $term_settings, $container_class);

Example:
add_filter(
    'ymc/filter/layout/left/custom_72_1',
    function ($output, $filter_id, $taxonomies, $term_settings, $container_class));

apply_filters(
    $hook_name,
    $output,              // фильтруемое значение (всегда первым)
    $filter_id,           // уникальный ID (идентификатор)
    $tax_name,            // контекст (таксономии)
    $term_settings,       // массив данных
    $container_class      // дополнительный параметр
);
